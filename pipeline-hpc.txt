
parallel 'hpc6-intel': {
    node('hpc6') {
        stage('HPC6 Clean') {
            if (params.BUILD_CLEAN) {
                cleanSetup()
            }
        }
        stage('HPC6 Update') {
            if (!params.BUILD_CLEAN) {
                updateSource()
            }
        }
        stage('HPC6 Configure') {
            sh 'mkdir -p opencmiss'
            dir('./setup') {
                git branch: 'develop', url: 'https://github.com/OpenCMISS/setup.git'
            }
            dir('./setup-build') {
                sh """#!/bin/bash
                  . /opt/intel/compilers_and_libraries_2017/linux/bin/compilervars.sh intel64
                  export FC=ifort
                  export CC=icc
                  export CXX=icc
                  export MPICH_FC=ifort
                  export MPICH_CC=icc
                  export MPICH_CXX=icc
                  export CMAKE_Fortran_COMPILER=ifort
                  export CMAKE_C_COMPILER=icc
                  export CMAKE_CXX_COMPILER=icc
                  /hpc_atog/cmiss/cmake-3.7.2-Linux-x86_64/bin/cmake -DOPENCMISS_ROOT=../opencmiss -DOPENCMISS_LIBRARIES=iron ../setup
                """
            }
        }
        stage('HPC6 Make') {
            dir('./setup-build') {
                sh """#!/bin/bash
                  . /opt/intel/compilers_and_libraries_2017/linux/bin/compilervars.sh intel64
                  export FC=ifort
                  export CC=icc
                  export CXX=icc
                  export MPICH_FC=ifort
                  export MPICH_CC=icc
                  export MPICH_CXX=icc
                  export CMAKE_Fortran_COMPILER=ifort
                  export CMAKE_C_COMPILER=icc
                  export CMAKE_CXX_COMPILER=icc
                  make
                """
            }
        }
        stage('HPC6 Test') {
            dir('./setup-build') {
                sh sh """#!/bin/bash
                  . /opt/intel/compilers_and_libraries_2017/linux/bin/compilervars.sh intel64
                  export FC=ifort
                  export CC=icc
                  export CXX=icc
                  export MPICH_FC=ifort
                  export MPICH_CC=icc
                  export MPICH_CXX=icc
                  export CMAKE_Fortran_COMPILER=ifort
                  export CMAKE_C_COMPILER=icc
                  export CMAKE_CXX_COMPILER=icc
                  make keytests
                """
            }
        }
    }
}, 'hpc5-mpich': {
    node('hpc5') {

                cleanSetup()
            }
        }
        stage('HPC5 Update') {
            if (!params.BUILD_CLEAN) {

            }
        }
        stage('HPC5 Configure') {

            }

        stage('HPC5 Make') {
