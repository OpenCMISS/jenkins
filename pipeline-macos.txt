
node('Mac10.12.3') {
    stage('Clean') {
        if (params.BUILD_CLEAN) {
            cleanSetup()
            sh 'rm -rf opencmiss'
        }
    }
    stage('Update') {
        if (!params.BUILD_CLEAN) {
            updateSource()
        }
    }
    stage('Configure') {
        sh 'mkdir -p opencmiss'
        dir('./setup') {
            git branch: 'develop', url: 'https://github.com/OpenCMISS/setup.git'
        }
        dir('./setup-build') {
            sh 'cmake -DCMAKE_C_FLAGS_RELEASE=-I/Library/Developer/CommandLineTools/SDKs/MacOSX.sdk/usr/include/ -DCMAKE_CXX_FLAGS_RELEASE=-I/Library/Developer/CommandLineTools/SDKs/MacOSX.sdk/usr/include/ -DOPENCMISS_LIBRARIES=iron -DOPENCMISS_ROOT=../opencmiss ../setup'
        }
    }
    stage('Make') {
        dir('./setup-build') {
        	sh 'make'
        }
    }
    stage('Test') {
        dir('./opencmiss/build/manage/release/config/Release') {
            sh 'make keytests'
        }
    }
    stage('Package') {
        dir('./opencmiss/build/manage/release/') {
            if (params.DO_RELEASE) {
                sh 'cmake -DOPENCMISS_RELEASE=ON .'
            } else {
                sh 'cmake -DOPENCMISS_RELEASE=OFF .'
            }
            sh 'make package_sdk'
        }
    }
    stage('Archive') {
        dir('./opencmiss/build/manage/release/packaging/sdk/package/') {
            archiveArtifacts artifacts: 'OpenCMISS-Libraries*'
        }
    }
    stage('Finish') {
        echo "Finish"
    }
}
